```js
// Array

//连接数组
Array.concat()

//将数组元素连接起来以构建一个字符串
Array.join()

//数组的大小
Array.length

//删除并返回数组的最后一个元素（末尾）
Array.pop()

//给数组添加元素（末尾）
Array.push()

//颠倒数组中元素的顺序
Array.reverse()

//将（头部）元素移出数组
Array.shift()

//返回数组的一部分
Array.slice()

//对数组元素进行排序
Array.sort()

//插入，删除，或替换数组的元素
Array.splice()

//把数组转换成局部字符串
Array.toLocaleString()

//将数组转换成一个字符串
Array.toString()

//在数组（头部）插入一个元素
Array.unshift()
```
